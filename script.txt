/*

Sistema de armazenamento em cache

-> No setor onde trabalho possuímos um site para o controle de mercadorias no estoque local

-> Toda vez que recarregamos o site, ele tem carregar o doom inteiro novamente, consequentemente,
as informações são perdidas e um custo computacional desnecessário e despendido para o recarregamento inteiro do DOOM

*/

HSET user:1 nomeUser "fulano" senha "1234" cache "https://site.com.br"

EXPIRE user:1 120 FIELDS 1 cache

/ * Salva as informações acerca do status do material */

/* Material é cadastado */ 
HSET material:1 rg "879654" quantidade 5

/* Material é removido parcialmente do estoque */
HINCRBY material:1 quantidade -1

/* Material é adcionado no estoque */
HINCRBY material:1 quantidade 20

/* Funcionarios recebem uma lista ordenada (define prioridade) de tarefas 
para serem realizadas na jornada de trabalho */

RPUSH tarefas-diarias "Fazer a reunião" "Consultar pendências" 
"Endereçar materias prioritários" "Endereçar outras pendẽncias" 
"Endereçar outros materiais" "Arrumar Setor" 

/* Conforme as tarefas são realizadas, elas são removidas da lista* /
LPOP tarefas-diais

/* Se o material se tornar obsoleto, ele poderá ser excluido do sistema */
DEL material:1
